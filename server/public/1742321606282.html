  1. В src создаем папку utils — тут будет функция, которая генерирует токен — создаем файлик generateToken.js и в него кладем:

require('dotenv').config();
const jwt = require('jsonwebtoken');
const jwtConfig = require('../configs/jwtConfig'); //! не забыть проверить пути

const { SECRET_ACCESS_TOKEN, SECRET_REFRESH_TOKEN } = process.env;

const generateToken = (payload) => ({ // payload - полезная нагрузка
  accessToken: jwt.sign(payload, SECRET_ACCESS_TOKEN, jwtConfig.access),
  refreshToken: jwt.sign(payload, SECRET_REFRESH_TOKEN, jwtConfig.refresh),
});

module.exports = generateToken;

  1. Создать папку middleware если еще нет или положить в имеющуюся мидлварку, которая будет проверять, есть ли у данного пользователя доступ — создаем файлик verifyToken.js (либо можно разнести их по разным файликам - verifyAccessToken и verifyRefreshToken, не забыть экспортировать) и внутрь кладем:

const jwt = require('jsonwebtoken');

const verifyRefreshToken = (req, res, next) => {
  try {
    const { refreshToken } = req.cookies; // достаем по ключу рефреш токен
    const { user } = jwt.verify(refreshToken, process.env.SECRET_REFRESH_TOKEN);
    res.locals.user = user;
    next();
  } catch (error) {
    console.log('Invalid refresh token');
    res.sendStatus(401);
  }
};

const verifyAccessToken = (req, res, next) => {
  try {
    const accessToken = req.headers.authorization.split(' ')[1];
    const { user } = jwt.verify(accessToken, process.env.SECRET_ACCESS_TOKEN);
    res.locals.user = user;
    next();
  } catch (error) {
    console.log('Invalid access token');
    res.sendStatus(401);
  }
};

module.exports = { verifyRefreshToken, verifyAccessToken };
  1. В папке routers создаем файлик token.router.js и в него кладем:

const router = require('express').Router();
const cookieConfig = require('../../configs/cookieConfig'); //! не забыть проверить пути
const { verifyRefreshToken } = require('../../middlewares/verifyToken'); //! не забыть проверить пути
const generateToken = require('../../utils/generateToken'); //! не забыть проверить пути

router.get('/refresh', verifyRefreshToken, (req, res) => {
  try {
    const { user } = res.locals;
    const { accessToken, refreshToken } = generateToken({ user });

    res
      .cookie('refreshToken', refreshToken, cookieConfig.refresh)
      .json({ user: user, accessToken });
  } catch (error) {
    console.error(error);
    res.sendStatus(400);
  }
});

module.exports = router;

router.post('/signup', async (req, res) => {
  const { username, email, password } = req.body;

  try {
    const [user, isCreated] = await User.findOrCreate({
      where: {
        email,
      },
      defaults: {
        username,
        email,
        password: await bcrypt.hash(password, 10),

    });

      res.status(400).json({ message: 'User alredy exist' });
    } else {
      const plainUser = user.get();
